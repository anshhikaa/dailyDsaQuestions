// problem : find building where alice and bob can meet
// plaform : leetcode - 2940
// difficulty : hard
// date : 11/2/26
// this is only the brute force approach 
#include <bits/stdc++.h>
using namespace std;
class Solution {
public:
    vector<int> leftmostBuildingQueries(vector<int>& heights, vector<vector<int>>& queries) {
        vector<int> ans;
        for(int i = 0 ;i<queries.size();i++){
            vector<int>v = queries[i];
            int apos = v[0];
            int bpos = v[1];
           if(apos == bpos){
            ans.push_back(apos);
            continue;
           }
           int left = min(apos,bpos);
           int right = max(apos,bpos);
           if(heights[right]>heights[left]){
                ans.push_back(right);
                continue;
            }
            int result = -1;
           int j = right +1 ;
            while(j<heights.size()){  
                if(heights[j] > heights[left] &&  heights[j] > heights[right]){
                    result = j ;
                    break;
                }
                else{
                    j++;
                }
            }
            ans.push_back(result);
        }
      return ans;  
    }
};
// optimised aprroach - monotonic stack, binary search
// date - 12/2/26
class Solution {
public:
    vector<int> leftmostBuildingQueries(vector<int>& heights, vector<vector<int>>& queries) {
        int n = heights.size();
        int m = queries.size();
        vector<int>ans(m,-1);
        vector<tuple<int,int,int>> offline; 
        for(int i = 0 ;i<m;i++){
            int a = queries[i][0];
            int b = queries[i][1];
            if(a==b){
                ans[i] = a;
                continue;
            }
            int left = min(a,b);
            int right = max(a,b);
            if(heights[right]>heights[left]){
                ans[i]= right;
            }
            else{
                offline.push_back({right,left,i});
            }
        }
        //sort queries
        sort(offline.begin(), offline.end(),[](auto &a, auto &b) {
                return get<0>(a) > get<0>(b);
     });

        vector<int>stack;
        int curr = n-1;
        for(auto&[right,left,indx]:offline){
            while(curr>right){
                while(!stack.empty() && heights[stack.back()] <= heights[curr]){
                    stack.pop_back();
                }
                stack.push_back(curr);
                curr--;
            }
            int threshold = max(heights[left],heights[right]);
            //binary search
            int l = 0 ;
            int r = stack.size()-1;
            int res = -1;
            while(l<=r){
                int mid = (l+r)/2;
                if(threshold < heights[stack[mid]]){
                    res=stack[mid];
                   l = mid+1;
                }
                else{
                    r= mid-1;
                }
            }
            ans[indx]= res;
        }
        return ans;
    }
};
