// problem : remove elements
// platform : leetcode (27)
// difficulty : easy
// t.c = o(n) , s.c = o(1);
#include<bits/stdc++.h>
using namespace std ;
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int index = 0 ;
        for(int i = 0 ;i<nums.size();i++){
            if(nums[i] != val){
                nums[index] = nums[i];
                index ++;
            }
        } 
        return index;
    }
};
